145 void Steady_insac::setBCs()
146 {
147  int i,j,k;
148  double vdotn;
149 
150  // boundary 0
151  i = m->gimx();
152  if(bcflags[0] == 1) // pressure outlet
153  for(j = 1; j <= m->gjmx()-1; j++)
154  u[0](i,j) = bvalues[0][0];
155  else if(bcflags[0] == 2) // no-slip wall
156  for(j = 1; j <= m->gjmx()-1; j++)
157  {
158  u[0](i,j) = u[0](i-1,j);
159  u[1](i,j) = 2.0*bvalues[0][0] - u[1](i-1,j);
160  u[2](i,j) = 2.0*bvalues[0][1] - u[2](i-1,j);
161  }
162  else // slip-wall
163  for(j = 1; j <= m->gjmx()-1; j++)
164  {
165  u[0](i,j) = u[0](i-1,j);
166  vdotn = u[1](i-1,j)*m->gdel(i-1,j,0) + u[2](i-1,j)*m->gdel(i-1,j,1);
167  u[1](i,j) = u[1](i-1,j) - 2*vdotn*m->gdel(i-1,j,0);
168  u[2](i,j) = u[2](i-1,j) - 2*vdotn*m->gdel(i-1,j,1);
169  }
170 
171  // boundary 2
172  i = 0;
173  if(bcflags[2] == 0) // parabolic velocity inlet
174  {
175  double a,b,c, rm;
176  rm = (m->gy(1,1) + m->gy(1,m->gjmx()))/2.0; // mid point
177  a = bvalues[2][0] / ( rm*rm - 2.0*rm*rm + m->gy(1,1)*m->gy(1,m->gjmx()) );
178  b = -a*2.0*rm;
179  c = -a*m->gy(1,1)*m->gy(1,1) - b*m->gy(1,1);
180  for(j = 1; j <= m->gjmx()-1; j++)
181  u[1](i,j) = a*m->gyc(i,j)*m->gyc(i,j) + b*m->gyc(i,j) + c;
182  }
183  else if(bcflags[2] == 1) // pressure outlet
184  for(j = 1; j <= m->gjmx()-1; j++)
185  u[0](i,j) = bvalues[2][0];
186  else if(bcflags[2] == 2) // no-slip wall
187  for(j = 1; j <= m->gjmx()-1; j++)
188  {
189  u[0](i,j) = u[0](i+1,j);
190  u[1](i,j) = 2.0*bvalues[2][0] - u[1](i+1,j);
191  u[2](i,j) = 2.0*bvalues[2][1] - u[2](i+1,j);
192  }
193  else // slip-wall
194  for(j = 1; j <= m->gjmx()-1; j++)
195  {
196  u[0](i,j) = u[0](i+1,j);
197  // we use negative of the del value as the normal always points in the positive i direction.
198  vdotn = u[1](i+1,j)*(-1*m->gdel(i,j,0)) + u[2](i+1,j)*(-1*m->gdel(i,j,1));
199  u[1](i,j) = u[1](i+1,j) - 2*vdotn*m->gdel(i+1,j,0);
200  u[2](i,j) = u[2](i+1,j) - 2*vdotn*m->gdel(i+1,j,1);
201  }
202 
203  // boundary 1
204  j = m->gjmx();
205  if(bcflags[1] == 1) // pressure outlet
206  for(i = 1; i <= m->gimx()-1; i++)
207  u[0](i,j) = bvalues[1][0];
208  else if(bcflags[1] == 2) // no-slip wall
209  for(i = 1; i <= m->gimx()-1; i++)
210  {
211  u[0](i,j) = u[0](i,j-1);
212  u[1](i,j) = 2*bvalues[1][0] - u[1](i,j-1);
213  u[2](i,j) = 2*bvalues[1][1] - u[2](i,j-1);
214  }
215  else if(bcflags[1] == 3) // slip wall
216  for(i = 1; i <= m->gimx()-1; i++)
217  {
218  u[0](i,j) = u[0](i,j-1);
219  vdotn = u[1](i,j-1)*m->gdel(i,j-1,2) + u[2](i,j-1)*m->gdel(i,j-1,3);
220  u[1](i,j) = u[1](i,j-1) - 2.0*vdotn*m->gdel(i,j-1,2);
221  u[2](i,j) = u[2](i,j-1) - 2.0*vdotn*m->gdel(i,j-1,3);
222  }
223 
224  // boundary 3
225  j = 0;
226  if(bcflags[3] == 1) // pressure outlet
227  for(i = 1; i <= m->gimx()-1; i++)
228  u[0](i,j) = bvalues[3][0];
229  else if(bcflags[3] == 2) // no-slip wall
230  for(i = 1; i <= m->gimx()-1; i++)
231  {
232  u[0](i,j) = u[0](i,j+1);
233  u[1](i,j) = 2*bvalues[3][0] - u[1](i,j+1);
234  u[2](i,j) = 2*bvalues[3][1] - u[2](i,j+1);
235  }
236  else if(bcflags[3] == 3) // slip wall
237  for(i = 1; i <= m->gimx()-1; i++)
238  {
239  u[0](i,j) = u[0](i,j+1);
240  // we use negative of the del values as the normal always points in the positive j-direction.
241  vdotn = u[1](i,j+1)*(-1)*m->gdel(i,j,2) + u[2](i,j+1)*(-1)*m->gdel(i,j,3);
242  u[1](i,j) = u[1](i,j+1) - 2.0*vdotn*m->gdel(i,j,2);
243  u[2](i,j) = u[2](i,j+1) - 2.0*vdotn*m->gdel(i,j,3);
244  }
245 
246  for(k = 0; k < nvar; k++)
247  {
248  u[k](0,0) = 0.5*(u[k](1,0)+u[k](0,1));
249  u[k](0,m->gjmx()) = 0.5*(u[k](1,m->gjmx())+u[k](0,m->gjmx()-1));
250  u[k](m->gimx(),0) = 0.5*(u[k](m->gimx(),1)+u[k](m->gimx()-1,0));
251  u[k](m->gimx(),m->gjmx()) = 0.5*(u[k](m->gimx()-1,m->gjmx())+u[k](m->gimx(),m->gjmx()-1));
252  }
253 }

258 void Steady_insac::compute_timesteps()
259 {
260  int i,j, dim;
261  vector<double> areavi(ndim), areavj(ndim); // area vectors
262  vector<double> inormal(ndim), jnormal(ndim); // unit normal vectors
263  double areai, areaj; // area magnitudes
264  double vdotni, vdotnj, eigeni, eigenj, voldt;
265  cout << "Steady_insac: compute_timesteps(): Now computing time steps for next iteration..." << endl;
266 
267  for(i = 1; i <= m->gimx()-1; i++)
268  for(j = 1; j <= m->gjmx()-1; j++)
269  {
270  for(dim = 0; dim < ndim; dim++) {
271  areavi[dim] = 0.5*(m->gdel(i,j,dim) + m->gdel(i-1,j,dim));
272  areavj[dim] = 0.5*(m->gdel(i,j,2+dim) + m->gdel(i-1,j,2+dim));
273  }
274  areai = sqrt(areavi[0]*areavi[0] + areavi[1]*areavi[1]);
275  areaj = sqrt(areavj[0]*areavj[0] + areavj[1]*areavj[1]);
276  for(dim = 0; dim < ndim; dim++)
277  {
278  inormal[dim] = areavi[dim]/areai;
279  jnormal[dim] = areavj[dim]/areaj;
280  }
281  // we now have face geometrical info "at the cell centers"
282 
283  vdotni = u[1](i,j)*inormal[0] * u[2](i,j)*inormal[1];
284  vdotnj = u[1](i,j)*jnormal[0] * u[2](i,j)*jnormal[1];
285 
286  eigeni = 0.5*( fabs(vdotni) + sqrt(vdotni*vdotni + 4.0*beta(i,j)*beta(i,j)) );
287  eigenj = 0.5*( fabs(vdotnj) + sqrt(vdotnj*vdotnj + 4.0*beta(i,j)*beta(i,j)) );
288  voldt = eigeni*areai + eigenj*areaj;
289  dt(i,j) = m->gvol(i,j)/voldt * cfl;
290  }
291 }

310 void Steady_insac::solve()
311 {
312  setInitialConditions();
313 
314  int i,j,k;
315  double resnorm, resnorm0, massflux;
316  cout << "Steady_insac: solve(): Beginning the time-march." << endl;
317  for(int n = 0; n < maxiter; n++)
318  {
319  // calculate stuff needed for this iteration
320  setBCs();
321  compute_beta();
322  compute_timesteps();
323 
324  // compute fluxes
325  for(k = 0; k < nvar; k++)
326  res[k].zeros();
327  invf->compute_fluxes();
328  grad->compute_fluxes();
329 
330  // check convergence
331  resnorm = 0; massflux = 0;
332  for(i = 1; i <= m->gimx()-1; i++)
333  for(j = 1; j <= m->gjmx()-1; j++)
334  {
335  resnorm += res[1].get(i,j)*res[1].get(i,j) + res[2].get(i,j)*res[2].get(i,j);
336  massflux += res[0].get(i,j);
337  }
338  if(n == 0) resnorm0 = resnorm;
339  if(n == 1 || n%10 == 0) cout << "Steady_insac: solve(): Iteration " << n << ":relative l2 norm of residual = " << resnorm/resnorm0 << ", net mass flux = " << massflux << endl;
340  if(resnorm/resnorm0 < tol && fabs(massflux) < sqrt(tol))
341  {
342  cout << "Steady_insac: solve(): Converged in " << n << " iterations. Norm of final residual = " << resnorm << ", final net mass flux = " << massflux << endl;
343  break;
344  }
345 
346  // update u
347  for(i = 1; i <= m->gimx()-1; i++)
348  for(j = 1; j <= m->gjmx()-1; j++)
349  {
350  u[0](i,j) = u[0].get(i,j) - res[0].get(i,j)*beta.get(i,j)*beta.get(i,j)*dt.get(i,j)/m->gvol(i,j);
351  for(k = 1; k < nvar; k++)
352  u[k](i,j) = u[k].get(i,j) - res[k].get(i,j)*dt.get(i,j)/(rho*m->gvol(i,j));
353  }
354  }
