176 void NormTanGradientIns::compute_s()
177 {
178  svect.resize(4);
179  for(int i = 0; i < 4; i++)
180  svect[i].setup(m->gimx()+1, m->gjmx()+1);

182  dels.resize(2);

184  dels[i].setup(m->gimx()+1, m->gjmx()+1);
185 
186  /* NOTE: We are not calculating svect and dels for the ghost cells i=imx+1 and j=jmx+1. */
187 
188  for(int i = 0; i <= m->gimx()-1; i++)
189  for(int j = 0; j <= m->gjmx()-1; j++)
190  {
191  svect[0](i,j) = m->gxc(i+1,j) - m->gxc(i,j);
192  svect[1](i,j) = m->gyc(i+1,j) - m->gyc(i,j);
193  dels[0](i,j) = sqrt(svect[0](i,j)*svect[0](i,j) + svect[1](i,j)*svect[1](i,j));
194  svect[0](i,j) /= dels[0](i,j);
195  svect[1](i,j) /= dels[0](i,j);
196 
197  svect[2](i,j) = m->gxc(i,j+1) - m->gxc(i,j);
198  svect[3](i,j) = m->gyc(i,j+1) - m->gyc(i,j);
199  dels[1](i,j) = sqrt(svect[2](i,j)*svect[2](i,j) + svect[3](i,j)*svect[3](i,j));
200  svect[2](i,j) /= dels[1](i,j);
201  svect[3](i,j) /= dels[1](i,j);
202  }
