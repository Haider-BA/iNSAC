141 void Poisson2d::compute_source()
142 {
143  // get source
144  cout << "Poisson2d: compute_source(): Computing source terms." << endl;
145  xcentre = 0; ycentre = 0;
146  double volsum = 0;
147  for(int i = 1; i <= m->gimx()-1; i++)
148  for(int j = 1; j <= m->gjmx()-1; j++)
149  {
150  volsum += m->gvol(i,j);
151  xcentre += m->gxc(i,j)*m->gvol(i,j);
152  ycentre += m->gyc(i,j)*m->gvol(i,j);
153  }
154  xcentre = xcentre/volsum;
155  ycentre = ycentre/volsum;
156 
157  for(int i = 1; i <= m->gimx()-1; i++)
158  for(int j = 1; j <= m->gjmx()-1; j++)
159  f(i,j) = source(m->gxc(i,j),m->gyc(i,j))*m->gvol(i,j);
160  cout << "Poisson2d: source terms computed" << endl;
161 }

184 void Poisson2d::setBCs()
185 {
186  //cout << "Poisson2d: setBCs(): Setting BCs" << endl;
187  double dn;
188  // dn is the perpendicular distance from cell center to boundary face.
189 
190  int j = 0;
191  if(bcflag[3] > 0)
192  for(int i = 1; i <= m->gimx()-2; i++)
193  {
194  dn = abs((m->gy(i+1,j+1)-m->gy(i,j+1))*m->gxc(i,j+1) - (m->gx(i+1,j+1)-m->gx(i,j+1))*m->gyc(i,j+1) + m->gx(i+1,j+1)*m->gy(i,j+1) - m->gy(i+1,j+1)*m->gx(i,j+1))
195  / sqrt((m->gy(i+1,j+1)-m->gy(i,j+1))*(m->gy(i+1,j+1)-m->gy(i,j+1)) + (m->gx(i+1,j+1)-m->gx(i,j+1))*(m->gx(i+1,j+1)-m->gx(i,j+1)));
196  u(i,j) = u(i,j+1) - 2*dn*bvalue[3];
197  }
198  else
199  for(int i = 1; i <= m->gimx()-2; i++)
200  u(i,j) = 2*bvalue[3] - u(i,j+1);
201 
202  j = m->gjmx();
203  if(bcflag[1] > 0)
204  for(int i = 1; i<=m->gimx()-2; i++)
205  {
206  dn = abs((m->gy(i+1,j)-m->gy(i,j))*m->gxc(i,j-1) - (m->gx(i+1,j)-m->gx(i,j))*m->gyc(i,j-1) + m->gx(i+1,j)*m->gy(i,j) - m->gy(i+1,j)*m->gx(i,j))
207  / sqrt((m->gy(i+1,j)-m->gy(i,j))*(m->gy(i+1,j)-m->gy(i,j)) + (m->gx(i+1,j)-m->gx(i,j))*(m->gx(i+1,j)-m->gx(i,j)));
208  u(i,j) = u(i,j-1) - 2*dn*bvalue[1];
209  }
210  else
211  for(int i = 1; i <= m->gimx()-2; i++)
212  u(i,j) = 2*bvalue[3] - u(i,j-1);
213 
214  int i = 0;
215  if(bcflag[2] > 0)
216  for(int j = 1; j <= m->gjmx()-2; j++)
217  {
218  dn = abs((m->gy(i+1,j+1)-m->gy(i+1,j))*m->gxc(i+1,j) - (m->gx(i+1,j+1)-m->gx(i+1,j))*m->gyc(i+1,j) + m->gx(i+1,j+1)*m->gy(i+1,j) - m->gy(i+1,j+1)*m->gx(i+1,j))
219  / sqrt((m->gy(i+1,j+1)-m->gy(i+1,j))*(m->gy(i+1,j+1)-m->gy(i+1,j)) + (m->gx(i+1,j+1)-m->gx(i+1,j))*(m->gx(i+1,j+1)-m->gx(i+1,j)));
220  u(i,j) = u(i+1,j) - 2*dn*bvalue[2];
221  }
222  else
223  for(int j = 1; j <= m->gjmx()-2; j++)
224  u(i,j) = 2*bvalue[2] - u(i+1,j);
225 
226  i = m->gimx();
227  if(bcflag[0] > 0)
228  for(int j = 1; j <= m->gjmx()-2; j++)
229  {
230  dn = abs((m->gy(i,j+1)-m->gy(i,j))*m->gxc(i-1,j) - (m->gx(i,j+1)-m->gx(i,j))*m->gyc(i-1,j) + m->gx(i,j+1)*m->gy(i,j) - m->gy(i,j+1)*m->gx(i,j))
231  / sqrt((m->gy(i,j+1)-m->gy(i,j))*(m->gy(i,j+1)-m->gy(i,j)) + (m->gx(i,j+1)-m->gx(i,j))*(m->gx(i,j+1)-m->gx(i,j)));
232  u(i,j) = u(i-1,j) - 2*dn*bvalue[0];
233  }
234  else
235  for(int j = 1; j <= m->gjmx()-2; j++)
236  u(i,j) = 2*bvalue[0] - u(i-1,j);
237 
238  // Corner ghost cells. Their values are just taken as average of the two neighboring ghost cells. This will hopefully not induce much error.

242  u(m->gimx(),m->gjmx()) = 0.5*(u(m->gimx(),m->gjmx()-1) + u(m->gimx()-1,m->gjmx()));

247 void Poisson2d::solve()
248 {
249  int k = 0;
250  double resnorm, resnorm0;
251  bool converged = false;
252 
253  do {
254  res.zeros();
255 
256  setBCs();
257 
258  g->compute_fluxes();
259 
260  //add source term
261  for(int i = 1; i < m->gimx()-1; i++)
262  for(int j = 1; j < m->gjmx()-1; j++)
263  res(i,j) -= f(i,j);
264 
265  // get resnorm
266  resnorm = 0;
267  for(int i = 1; i <= m->gimx()-1; i++)
268  for(int j = 1; j < m->gjmx()-1; j++)
269  resnorm += res.get(i,j)*res.get(i,j);
270  resnorm = sqrt(resnorm);
271  if(k == 0) resnorm0 = resnorm;
272  resnorms.push_back(resnorm/resnorm0);
273 
274  if(resnorm/resnorm0 < tol)
275  {
276  cout << "Poisson2d: solve(): Solver converged in " << k << " iterations." << endl;
277  cout << "Poisson2d: solve(): Final area-weighted residual is " << resnorm*(1.0/(m->gimx()-1)) << endl;
278  converged = true;
279  break;
280  }
281 
282  //update
283  solver->update();
284 
285  if(k%10 == 0)
286  cout << "Poisson2d: solve(): Iteration " << k << ", residual = " << resnorm/resnorm0 << endl;
287  k++;
288  } while(k < maxiter);
289 
290  if(!converged) {
291  cout << "Poisson2d: solve(): Solver could not converge!!" << endl;
292  }
293 }
