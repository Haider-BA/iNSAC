16 double minmod_avg(double a, double b)
17 {
18  double signa, signb;
19  if(fabs(a) <= ZERO_TOL) signa = 0;
20  else signa = a/fabs(a);
21  if(fabs(b) <= ZERO_TOL) signb = 0;
22  else signb = b/fabs(b);
23 
24  if(fabs(a) <= fabs(b))
25  return (signa+signb)*0.5*fabs(a);
26  else
27  return (signa+signb)*0.5*fabs(b);
28 }

56 void BasicPR::compute_pressure_difference()
57 {
58  // we include the i=0 and j=0 ghost cells
59  for(int j = 0; j <= m->gjmx()-1; j++)
60  for(int i = 0; i <= m->gimx()-1; i++)
61  {
62  dp[0](i,j) = u[0].get(i,j) - u[0].get(i+1,j); // p_L - p_R
63  dp[1](i,j) = u[0].get(i,j) - u[0].get(i,j+1); // p_L - p_R in j-direction, so it's more like p_down - p_up
64  }

75 void TVDPR::compute_pressure_difference()
76 {
77  int i,j;
78  for(j = 1; j <= m->gjmx()-2; j++)
79  for(i = 1; i <= m->gimx()-2; i++)
80  {
81  dp[0](i,j) = u[0].get(i,j) +0.5*minmod_avg(u[0].get(i+1,j)-u[0](i,j), u[0].get(i,j)-u[0].get(i-1,j)); // p_L
82  dp[0](i,j) -= u[0].get(i+1,j) - 0.5*minmod_avg(u[0].get(i+1,j)-u[0].get(i,j), u[0].get(i+2,j)-u[0].get(i+1,j)); // p_R
83  dp[1](i,j) = u[0].get(i,j) +0.5*minmod_avg(u[0].get(i,j+1)-u[0](i,j), u[0].get(i,j)-u[0].get(i,j-1)); // p_down
84  dp[1](i,j) -= u[0].get(i,j+1) - 0.5*minmod_avg(u[0].get(i,j+1)-u[0].get(i,j), u[0].get(i,j+2)-u[0].get(i,j+1)); // p_up
85  }
86 
87  // Now values for remaining cells (real or ghost) - just copy values from adjacent interior cells.
88  // We do not need values for cells i = imx and j = jmx (ghost cells). We also probably don't need dp for corner cells.
89  for(j = 1; j <= m->gjmx()-2; j++)
90  {
91  dp[0](0,j) = dp[0](1,j);
92  dp[0](m->gimx()-1,j) = dp[0](m->gimx()-2,j);
93  dp[1](0,j) = u[0].get(0,j) +0.5*minmod_avg(u[0].get(0,j+1)-u[0](0,j), u[0].get(0,j)-u[0].get(0,j-1)); // p_down
94  dp[1](0,j) -= u[0].get(0,j+1) - 0.5*minmod_avg(u[0].get(0,j+1)-u[0].get(0,j), u[0].get(0,j+2)-u[0].get(0,j+1)); // p_up
95  dp[1](m->gimx()-1,j) = u[0].get(,j) +0.5*minmod_avg(u[0].get(,j+1)-u[0](,j), u[0].get(,j)-u[0].get(,j-1));
96  dp[1](m->gimx()-1,j) -= u[0].get(m->gimx()-1,j+1) - 0.5*minmod_avg(u[0].get(m->gimx()-1,j+1)-u[0].get(m->gimx()-1,j), u[0].get(m->gimx()-1,j+2)-u[0].get(m->gimx()-1,j+1));
97  }
98  for(i = 1; i <= m->gimx()-2; i++)
99  {
100  dp[1](i,0) = dp[1](i,1);
101  dp[1](i,m->gjmx()-1) = dp[1](i,m->gjmx()-2);
102  dp[0](i,0) = u[0].get(i,0) +0.5*minmod_avg(u[0].get(i+1,0)-u[0](i,0), u[0].get(i,0)-u[0].get(i-1,0)); // p_L
103  dp[0](i,0) -= u[0].get(i+1,0) - 0.5*minmod_avg(u[0].get(i+1,0)-u[0].get(i,0), u[0].get(i+2,0)-u[0].get(i+1,0)); // p_R
104  dp[0](i,m->gjmx()-1) = u[0].get(i,j) +0.5*minmod_avg(u[0].get(i+1,j)-u[0](i,j), u[0].get(i,j)-u[0].get(i-1,j));
105  dp[0](i,m->gjmx()-1) -= u[0].get(i+1,m->gjmx()-1) - 0.5*minmod_avg(u[0].get(i+1,m->gjmx()-1)-u[0].get(i,m->gjmx()-1), u[0].get(i+2,m->gjmx()-1)-u[0].get(i+1,m->gjmx()-1));
106  }

187 void InviscidFlux::compute_flux()
188 {
189  pr->compute_pressure_difference();
190 
191  // account for solid walls.
192  vector<int> wall(4); // one flag for each of the 4 boundaries.
193  for(int i = 0; i < 4; i++)
194  if(bcflags[i] == 2 || bcflags[i] == 3)
195  wall[i] = 0;
196  else
197  wall[i] = 1;
198 
199  // add inviscid flux contribution to residuals
200  cout << "InviscidFlux: compute_flux(): Computing inviscid fluxes now..." << endl;
201  // We consider the first ghost cell layer for i, but not the last.
202  int i, j, k;
203  double area, nx, ny, eigen, vdotn;
204  double bhalf; vector<double> uhalf(3); // interface values for each unknown
205  Array2d<double> g(m->gimx(),nvar);
206 
207  for(j = 1; j <= m->gjmx()-1; j++)
208  {
209  for(i = 0; i <= m->gimx()-1; i++)
210  {
211  area = sqrt(m->gdel(i,j,0)*m->gdel(i,j,0) + m->gdel(i,j,1)*m->gdel(i,j,1));
212  nx = m->gdel(i,j,0)/area;
213  ny = m->gdel(i,j,1)/area;
214  for(k = 0; k < nvar; k++)
215  uhalf[k] = 0.5*(u[k](i,j) + u[k](i+1,j));
216  bhalf = 0.5*(beta->get(i,j) + beta->get(i+1,j);
217  vdotn = uhalf[1]*nx + uhalf[2]*ny;
218  // now get eigenvalue for Rhie-Chow
219  eigen = 0.5*( fabs(vdotn) + sqrt(vdotn*vdotn + 4*bhalf*bhalf) );
220 
221  // Boundary faces need to be treated differently to account for wall BCs.
222  if(i == 0)
223  {
224  g(i,0) = area*(rho*vdotn + c*eigen*dp[0](i,j)/(bhalf*bhalf))*wall[2];
225  g(i,1) = area*(rho*vdotn*uhalf[1]*wall[2] + uhalf[0]*nx);
226  g(i,2) = area*(rho*vdotn*uhalf[2]*wall[2] + uhalf[0]*ny);
227  }
228  else if(i == m->gimx()-1)
229  {
230  g(i,0) = area*(rho*vdotn + c*eigen*dp[0](i,j)/(bhalf*bhalf))*wall[0];
231  g(i,1) = area*(rho*vdotn*uhalf[1]*wall[0] + uhalf[0]*nx);
232  g(i,2) = area*(rho*vdotn*uhalf[2]*wall[0] + uhalf[0]*ny);
233  }
234  else
235  {
236  g(i,0) = area*(rho*vdotn + c*eigen*dp[0](i,j)/(bhalf*bhalf));
237  g(i,1) = area*(rho*vdotn*uhalf[1] + uhalf[0]*nx);
238  g(i,2) = area*(rho*vdotn*uhalf[2] + uhalf[0]*ny);
239  }
240  }
241  for(i = 1; i <= m->gimx()-1; i++)
242  for(k = 0; k < nvar; k++)
243  res[k](i,j) += g(i,k) - g(i-1,k);
244  }
245 
246  // now we add contribution of j-fluxes
247  Array2d<double> h(m->gjmx(),nvar);
248 
249  for(i = 1; i <= m->gimx()-1; i++)
250  {
251  for(j = 0; j <= m->gjmx()-1; j++)
252  {
253  area = sqrt(m->gdel(i,j,2)*m->gdel(i,j,2) + m->gdel(i,j,3)*m->gdel(i,j,3));
254  nx = m->gdel(i,j,2)/area;
255  ny = m->gdel(i,j,3)/area;
256  for(k = 0; k < nvar; k++)
257  uhalf[k] = 0.5*(u[k](i,j) + u[k](i,j+1));
258  bhalf = 0.5*(beta->get(i,j) + beta(i,j+1));
259  vdotn = uhalf[1]*nx + uhalf[2]*ny;
260  eigen = 0.5*( fabs(vdotn) + sqrt(vdotn*vdotn + 4*bhalf*bhalf) );
261 
262  if(j == 0)
263  {
264  h(j,0) = area*(rho*vdotn + c*eigen*dp[1](i,j)/(bhalf*bhalf))*wall[3];
265  h(j,1) = area*(rho*vdotn*uhalf[1]*wall[3] + uhalf[0]*nx);
266  h(j,2) = area*(rho*vdotn*uhalf[2]*wall[3] + uhalf[0]*ny);
267  }
268  else if(j == m->gjmx()-1)
269  {
270  h(j,0) = area*(rho*vdotn + c*eigen*dp[1](i,j)/(bhalf*bhalf))*wall[1];
271  h(j,1) = area*(rho*vdotn*uhalf[1]*wall[1] + uhalf[0]*nx);
272  h(j,2) = area*(rho*vdotn*uhalf[2]*wall[1] + uhalf[0]*ny);
273  }
274  else
275  {
276  h(j,0) = area*(rho*vdotn + c*eigen*dp[1](i,j)/(bhalf*bhalf));
277  h(j,1) = area*(rho*vdotn*uhalf[1] + uhalf[0]*nx);
278  h(j,2) = area*(rho*vdotn*uhalf[2] + uhalf[0]*ny);
279  }
280  }
281  for(j = 1; j < m->gjmx()-1; j++)
282  for(k = 0; k < nvar; k++)
283  res[k](i,j) += h(j,k) - h(j-1,k);
284  }
285 }
